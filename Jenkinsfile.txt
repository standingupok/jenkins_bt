pipeline {
    agent any
    environment {
        VENV = 'venv' // Define virtual environment directory
    }
    stages {
        stage('Setup Environment') {
            steps {
                script {
                    def password = credentials('sudo-jenkins-password')
                    sh """
                        echo $password | sudo -S apt update
                        sudo apt install python3-venv
                    """
                }
            }
        }
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/standingupok/jenkins_bt.git'
            }
        }
        stage('Install Dependencies') {
            steps {
                sh """
                python3 -m venv $VENV          # Create virtual environment
                source $VENV/bin/activate      # Activate virtual environment
                pip install --upgrade pip      # Upgrade pip
                pip install -r requirements.txt # Install dependencies
                """
            }
        }
        stage('Run Application') {
            steps {
                sh """
                source $VENV/bin/activate
                uvicorn main:app --host 127.0.0.1 --port 8000 &
                """
            }
        }
        stage('Run Tests') {
            steps {
                sh """
                source $VENV/bin/activate
                pytest test_app.py --maxfail=1 --disable-warnings
                """
            }
        }
    }
    post {
        always {
            sh 'rm -rf $VENV' // Cleanup virtual environment
        }
    }
}
