pipeline {
    agent any
    environment {
        VENV = 'venv' // Define a variable for the virtual environment directory
    }
    
    stages {
        stage('Setup Environment') {
            steps {
                sh '''
                sudo apt update
                sudo apt install -y python3-venv
                '''
            }
        }
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/standingupok/jenkins_bt.git'
            }
        }
        stage('Install Dependencies') {
            steps {
                sh '''
                python3 -m venv $VENV         # Create a virtual environment
                source $VENV/bin/activate     # Activate the virtual environment
                pip install --upgrade pip     # Upgrade pip
                pip install -r requirements.txt  # Install dependencies
                '''
            }
        }
        stage('Run Application') {
            steps {
                sh '''
                source $VENV/bin/activate     # Activate the virtual environment
                uvicorn main:app --host 127.0.0.1 --port 8000 &
                '''
            }
        }
        stage('Run Tests') {
            steps {
                sh '''
                source $VENV/bin/activate     # Activate the virtual environment
                pytest test_app.py --maxfail=1 --disable-warnings
                '''
            }
        }
    }
    post {
        always {
            sh 'rm -rf $VENV'  // Cleanup the virtual environment after pipeline execution
        }
    }
}
